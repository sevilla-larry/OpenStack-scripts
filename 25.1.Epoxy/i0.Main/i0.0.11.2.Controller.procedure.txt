        ### modify "stop" of /etc/init.d/mariadb

        nano /etc/init.d/mariadb

        # Recommended by Grok 2025/Oct/30
        # due to SQL errors during reboot or poweroff
        #

        stop)
            log_info_msg "Stopping MariaDB daemon..."

            # Wait for HTTPD to fully stop first (dependency)
            WAIT_TIMEOUT=30  # Max seconds to wait
            WAIT_COUNT=0
            while pidof httpd >/dev/null && [ $WAIT_COUNT -lt $WAIT_TIMEOUT ]; do
                log_info_msg "Waiting for HTTPD to stop..."
                sleep 1
                WAIT_COUNT=$((WAIT_COUNT + 1))
            done

            if [ $WAIT_COUNT -eq $WAIT_TIMEOUT ]; then
                log_warning_msg "Timeout waiting for HTTPD — proceeding anyway."
            fi

            killproc -p ${PIDFILE} /usr/sbin/mariadbd
            evaluate_retval
            ;;

        # MariaDB end of stop)


        ### modify "stop" of /etc/init.d/httpd

        nano /etc/init.d/httpd

        # Recommended by Grok 2025/Oct/30
        # due to SQL errors during reboot or poweroff
        #

        stop)
            log_info_msg "Stopping Apache HTTP daemon..."
            start_daemon /usr/sbin/apachectl -k graceful-stop
            evaluate_retval
            ;;

        # httpd end of stop)
