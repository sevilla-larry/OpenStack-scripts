A. Preparation

	a. update /etc/hostname							controller
	b. update /etc/sysconfig/rc.site (hostname)		controller
	c. update networks with the ff:

		1. ifconfig.eth0		### Management NAT
			ONBOOT=yes
			IFACE=eth0
			SERVICE=ipv4-static
			IP=10.0.0.11
			GATEWAY=10.0.0.1
			PREFIX=24
			BROADCAST=10.0.0.255

	    2. ifconfig.eth1		### Provider Open
			ONBOOT=yes
			IFACE=eth1
			SERVICE=bridge

	d.  check /lib/services/bridge

	e. # nano /root/.bashrc

		export PATH=/usr/local/bin:/usr/local/sbin:/usr/bin:/usr/sbin

	f. $ nano /home/user/.bashrc

		export PATH=/usr/local/bin:/usr/bin

	g. check /etc/hosts
